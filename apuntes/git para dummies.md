# Bajarse un repo
git clone https://github.com/ivanCeballos-upgradeHub/202311-javascript

# Ver estado del repo
git status

# Ver las ramas que tengo
git branch

# Cambiar a una rama
git checkout <rama>

# Crear una rama
git checkout -b <rama>

# Push al repositorio
git push
git push origin <nombre de rama en remoto>
git push --set-upstream origin <nombre de rama en remoto>
